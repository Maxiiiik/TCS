# Встановлюємо робочу директорію в контейнері
WORKDIR /app

# Запускаємо необхідні команди для налаштування проекту
RUN command1 && command2

# Вказуємо команду, яка буде виконуватися при запуску контейнера
CMD ["command", "arg1", "arg2"]
